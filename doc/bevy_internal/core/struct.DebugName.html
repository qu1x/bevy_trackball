Expand description

Convenient query for giving a human friendly name to an entity.

fn increment_score(mut scores: Query<(DebugName, &mut Score)>) {
    for (name, mut score) in &mut scores {
        score.0 += 1.0;
        if score.0.is_nan() {
            bevy_utils::tracing::error!("Score for {:?} is invalid", name);
        }
    }
}
